  def to_o(v)
    if v == nil
      return nil
    end
    s = v.to_s
    if not s
      return nil
    end
    if s == 'nil'
      return nil
    end
    if s =~ /^\d+$/
      return s.to_i
    end
    if s =~ /^\d+\.?\d*$/
      return s.to_f
    end
    if s == 'true'
      return true
    end
    if s == 'false'
      return false
    end
    if s =~ /^\[.*\]$/
      s = s.gsub /^\[/,''
      s = s.gsub /\]$/,''
      return s.split(',').map {|x| to_o x}
    end
    return s
  end
