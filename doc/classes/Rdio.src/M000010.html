  def self.api
    return @api if @api
    key = ENV[RDIO_KEY]
    secret = ENV[RDIO_SECRET]
    if key and secret
      @api = init key,secret
    else
      msg = "Could not construct an Api instance.  You must first call init(key,secret)\nor set the following environmental variables:\n- \#{RDIO_KEY}\n- \#{RDIO_SECRET}\n"
      raise Exception.new msg
    end
    return @api
  end
